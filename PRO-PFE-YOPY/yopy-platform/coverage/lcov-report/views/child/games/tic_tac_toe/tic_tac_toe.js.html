    function findWinningMove(sym) {
        for(let line of WIN_LINES) {

    function getWinner(b) {

    }
    function minimax(b, turn, depth=0) {
        let w = getWinner(b);
        if(w==='O') return {score:10-depth};
        if(w==='X') return {score:-10+depth};
        let empty = b.reduce((arr,val,i)=> val===null ? [...arr,i] : arr, []);
        if(empty.length===0) return {score:0};
        let moves = [];
        for(let idx of empty) {
            let nb = [...b]; nb[idx]=turn;
            let res = minimax(nb, turn==='O'?'X':'O', depth+1);
            moves.push({index:idx, score:res.score});
        }
        if(turn==='O') return moves.reduce((a,m)=> m.score>a.score ? m : a);
        else return moves.reduce((a,m)=> m.score<a.score ? m : a);
    }
    function pickBotMove() {
        let empty = board.map((v,i)=> v===null ? i : null).filter(v=>v!==null);

        if(difficulty==='medium') {
            let win = findWinningMove('O'); if(win!==null) return win;
            let block = findWinningMove('X'); if(block!==null) return block;
            return empty[Math.floor(Math.random()*empty.length)];
        }
        let best = minimax(board,'O');
        return best.index;
    }

    let result = checkResult();
    if (result) {
        await handleRoundEnd(result);
        return;
    }

    playerTurn = false;
    lockBoard(true);
    setTurnMsg('🍃 AI thinking...');
    setTimeout(() => botMoveAsync(), DIFF_CONFIG[difficulty].botDelay);
}

   async function handleRoundEnd(result) {
    roundOver = true;
    lockBoard(true);
    if (result.winner === 'X') {
        highlightWin(result.line, 'X');
        scoreYou++;
        sessionScore += DIFF_CONFIG[difficulty].mult;
        if (collector) collector.record('win', null);
        setTurnMsg('✨ Brilliant! You win ✨');
    } else if (result.winner === 'O') {
        highlightWin(result.line, 'O');
        scoreBot++;
        if (collector) collector.record('lose', null);
        setTurnMsg('🍂 The AI wins...');
    } else {
        if (difficulty === 'hard') sessionScore += Math.floor(DIFF_CONFIG.hard.mult / 2);
        scoreTie++;
        if (collector) collector.record('tie', null);
        setTurnMsg('🤝 Tie · Balanced mind');
    }
    updateScores();
    round++;
    showRoundResultPopup(result);
}

async function endSession() {
    if (gameState !== STATE.PLAYING) return;
    gameState = STATE.DONE;

    if (collector) {
        await collector.flush();
    }

    let elapsed = gameStartTime ? Math.max(3, Math.floor((Date.now() - gameStartTime) / 1000)) : 3;

    let emoji, title;
    if (scoreYou > scoreBot) { emoji = '🏆 Victory'; title = 'Masterful Stroke'; }
    else if (scoreBot > scoreYou) { emoji = '🍂 Reflection'; title = 'Quiet Insight'; }
    else { emoji = '☯️ Harmony'; title = 'Balanced Spirit'; }

    document.getElementById('result-emoji').innerHTML = emoji;
    document.getElementById('result-title').innerText = title;
    document.getElementById('result-score').innerHTML = `Score: ${sessionScore}`;
    let diffName = difficulty === 'easy' ? 'Easy' : (difficulty === 'medium' ? 'Medium' : 'Hard');
    document.getElementById('result-detail').innerHTML = `You ${scoreYou} : ${scoreBot} AI · Ties ${scoreTie} · ${diffName}`;
    document.getElementById('result-overlay').style.display = 'flex';

    await submitScore(sessionScore, elapsed);
    document.getElementById('round-result-overlay').style.display = 'none';
}

    (function inkParticles() {
        const canvas = document.getElementById('ink-canvas');
        if(!canvas) return;
        const ctx = canvas.getContext('2d');
        let particles = [];
        function resize() { canvas.width = window.innerWidth; canvas.height = window.innerHeight; }
        resize(); window.addEventListener('resize', resize);
        for(let i=0;i<70;i++) particles.push({ x:Math.random()*canvas.width, y:Math.random()*canvas.height, r:Math.random()*4+1.5, alpha:Math.random()*0.3+0.1, vx:(Math.random()-0.5)*0.2, vy:(Math.random()-0.5)*0.1 });
        function draw() {
            if(!ctx) return;
            ctx.clearRect(0,0,canvas.width,canvas.height);
            particles.forEach(p=>{
                ctx.beginPath(); ctx.arc(p.x,p.y,p.r,0,Math.PI*2);
                ctx.fillStyle = `rgba(70, 45, 28, ${p.alpha})`;
                ctx.fill();
                p.x += p.vx; p.y += p.vy;
                if(p.x<0) p.x=canvas.width; if(p.x>canvas.width) p.x=0;
                if(p.y<0) p.y=canvas.height; if(p.y>canvas.height) p.y=0;
            });
            requestAnimationFrame(draw);
        }
        if(!window.matchMedia('(prefers-reduced-motion: reduce)').matches) draw();
    })();
